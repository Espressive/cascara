A data module to display **string** values, it renders a input of type **text**.

### Playground

Feel free to play around with the following code:

```jsx
<ModuleExample
  isEditing={true}
  record={{
    id: 1,
    firstName: 'John',
    lastName: 'Doe',
  }}
>
  <DataText
    attribute='firstName'
    isEditable={true}
    isLabeled={true}
    label='First Name'
    value='John'
  />
</ModuleExample>
```

### Use in Table

Adding this module to your **Cascara** **Table** or **Form** would look like this:

```jsx
<Table
  // we need to enable edition by passing the `edit` module
  actions={{
    actionButtonMenuIndex: -1,
    modules: [
      {
        module: 'edit',
        cancelLabel: 'Cancelar',
        editLabel: 'Editar',
        saveLabel: 'Guardar',
      },
    ],
  }}
  // your data
  data={[
    {
      id: 1,
      firstName: 'John',
    },
  ]}
  // the fields in our data you want to display
  dataConfig={{
    display: [
      {
        attribute: 'firstName',
        isEditable: true,
        isLabeled: false,
        label: 'First Name',
        module: 'text',
      },
    ],
  }}
  // the action handler is invoken when a button is clicked
  onAction={(action, record) => {
    console.clear();
    console.log(`Action ${action.name} invoked`);
    console.table(record);

    switch (action.name) {
      case 'edit.start':
        console.log('Entering edit mode');
        break;
      case 'edit.cancel':
        console.log('Exiting edit mode, edit canceled');
        break;
      case 'edit.save':
        console.log('Exitin edit mode, saving changes');
        break;
    }
  }}
  // this defines the unique ID field in your data
  uniqueIdAttribute={'id'}
/>
```
