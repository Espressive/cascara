A data module to display multiple option values, it renders a select element.

### Playground

Feel free to play around with the following code:

```jsx
<ModuleExample
  isEditing={true}
  record={{
    id: 1,
    ki: 45,
  }}
>
  <DataSelect
    attribute='color'
    isEditable={true}
    isLabeled={true}
    label='Color'
    options={[
      {
        key: 'blue',
        value: 'blue',
        text: 'Blue',
      },
      {
        key: 'red',
        value: 'red',
        text: 'Red',
      },
      {
        key: 'white',
        value: 'white',
        text: 'White',
      },
    ]}
    value='blue'
  />
</ModuleExample>
```

### Use in Table

Adding this module to your **Cascara** **Table** or **Form** would look like this:

```jsx
<Table
  // we need to enable edition by passing the `edit` module
  actions={{
    actionButtonMenuIndex: -1,
    modules: [
      {
        module: 'edit',
        cancelLabel: 'Cancelar',
        editLabel: 'Editar',
        saveLabel: 'Guardar',
      },
    ],
  }}
  // your data
  data={[
    {
      id: 1,
      color: 'blue',
    },
  ]}
  // the fields in our data you want to display
  dataConfig={{
    display: [
      {
        attribute: 'color',
        isEditable: true,
        isLabeled: false,
        label: 'Color',
        module: 'select',
        options: [
          {
            key: 'blue',
            value: 'blue',
            text: 'Blue',
          },
          {
            key: 'red',
            value: 'red',
            text: 'Red',
          },
          {
            key: 'white',
            value: 'white',
            text: 'White',
          },
        ],
      },
    ],
  }}
  // the action handler is invoken when a button is clicked
  onAction={(action, record) => {
    console.clear();
    console.log(`Action ${action.name} invoked`);
    console.table(record);

    switch (action.name) {
      case 'edit.start':
        console.log('Entering edit mode');
        break;
      case 'edit.cancel':
        console.log('Exiting edit mode, edit canceled');
        break;
      case 'edit.save':
        console.log('Exitin edit mode, saving changes');
        break;
    }
  }}
  // this defines the unique ID field in your data
  uniqueIdAttribute={'id'}
/>
```
